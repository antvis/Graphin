register?: {
    /** G6 */
    nodeShape?: (G6: G6Type) => Register[];
    /** G6 */
    edgeShape?: (G6: G6Type) => Register[];
    /** G6 */
    behavior?: (G6: G6Type) => BehaviorRegister[];
};
interface Register {
    /**  */
    name: string;
    /** register,G6 */
    register: (G6: G6Type) => void;
}
interface BehaviorRegister extends Register {
    options: any;
    mode: string;
}

register: {
    nodeShape: (G6) => [{
        name: "custom",
        register: () => {
            G6.registerNode("custom", () => {})  //  G6 registerNode 
        }
    }],
    edgeShape: (G6) => [{
        name: "custom",
        register: () => {
            G6.registerEdge("custom", () => {})  //  G6 registerEdge 
        }
    }],
    behavior: (G6) => [{
        mode: "default",  //  G6  mode 
        options: {}
        name: "custom",
        register: () => {
            G6.registerBehavior("custom", () => {})  //  G6 registerBehavior 
        }
    }],
}
